
#include <bits/stdc++.h>
using namespace std;
#define lint long long int
int main() {
    lint t;
    cin>>t;
    while(t--){
        lint n;
        cin>>n;
        lint a[n];
        cin>>a[0];
        for(lint i=1;i<n;i++){
            cin>>a[i];
            if(a[i]<a[i-1]){
                cout<<a[i]<<" ";
            }
            else{
                cout<<-1<<" ";
            }
        }
        cout<<-1<<endl;
    }
	
}
